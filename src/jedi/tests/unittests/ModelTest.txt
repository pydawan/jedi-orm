	@Test
	public void testInsert() {
		
		String esperado = "Zimbabwe";
		Pais obtido = Pais.objects.get("sigla", "ZW");
		
		Assert.assertTrue(esperado.equals(obtido.nome() ) );
	}
	
	@Test
	public void testUpdate() {
		
		Pais esperado = Pais.objects.get("sigla", "YT");
		esperado.update("nome='MAYOTTE'");		
		Pais obtido = Pais.objects.get("sigla", "YT");
		
		Assert.assertTrue(esperado.nome().equals(obtido.nome() ) );		
	}
	
	@Test
	public void testDelete() {
		
		int esperado = Pais.objects.count();
		
		Pais.objects.filter("sigla='YT'").first().delete();
		
		int obtido = esperado - 1;
		
		Assert.assertEquals(esperado, obtido + 1);
	}
	
	@Test
	public void testSaveInsert() {		
		
		Pais esperado = new Pais("Your Country", "YC");
		esperado.save();
		
        // Adiciona aos paises para que em testCleanup o registro 
		// possa ser excluído da tabela.		
		paises.add(esperado);
		
		Pais obtido = Pais.objects.get("sigla", "YC");
		
		Assert.assertEquals(esperado, obtido);
	}
	
	@Test
	public void testSaveUpdate() {
		
		String esperado = "Mayotte";
		
		Pais pais = Pais.objects.get("sigla", "YT");
		pais.nome(esperado);
		
		String obtido = pais.nome();
		
		Assert.assertTrue(esperado.equals(obtido) );
